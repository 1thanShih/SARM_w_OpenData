import pandas as pd
df = pd.read_parquet('/tmp/sarm_materialize/snapshot/meta/episodes/chunk-000/file-000.parquet')
print(f'total rows: {len(df)}')
# 驗證 dense 標注覆蓋率
import numpy as np
cov = []
for i in range(len(df)):
    row = df.iloc[i]
    length = int(row['length'])
    end_f = list(row['dense_subtask_end_frames'])
    cov.append(end_f[-1] / length if end_f else 0)
arr = np.array(cov)
print(f'dense coverage: min={arr.min():.3f} mean={arr.mean():.3f}')
assert arr.min() >= 0.95, '有 episode 覆蓋率 < 95%，annotations 可能有問題'
print('✓ 全部 episode dense 覆蓋率 ≥ 95%')
EOF
過關條件 — sanity check 看到三個 0 NULL/empty + spot check 印出 min ≥ 0.999 才能進 Step 3。

Step 4 — Hub 端驗證

從 Hub 重新下載 parquet 並斷言 0 NULL， 確認線上版本真的是新的（不是本地 cache）。

4

強制刷新下載 + assert

2 分鐘
為什麼：hf_hub_download 預設會用本地 cache，可能拿到 push 前的舊版本。force_download=True 跟 revision='v3.0' 一起用，確認 Hub 上的 v3.0 tag 真的指向修好的 commit。
python3 <<'EOF'
import pandas as pd
from huggingface_hub import hf_hub_download

p = hf_hub_download(
    'Lebruhbruh/SARM-opendata-annotated-fixed',
    'meta/episodes/chunk-000/file-000.parquet',
    repo_type='dataset',
    revision='v3.0',
    force_download=True,
)
df = pd.read_parquet(p)
empty = sum(1 for v in df['dense_subtask_names']
            if v is None or (hasattr(v, '__len__') and len(v) == 0))
print(f'dense_subtask_names empty: {empty}/{len(df)}')
assert empty == 0, '⚠ Hub 上還是有空的 dense_subtask_names'

# 也順便看一筆中段 episode 的內容
import json

print('✓ Hub v3.0 確認完整')
EOF
過關條件 — 看到 dense_subtask_names empty: 0/273 + 中段 episode 的 dense_end_frames 看起來合理（最後一個 frame ≈ episode length）。

Rollback

如果 Step 3 push 後發現腳本邏輯有錯，要把 Hub 還原成 push 前的狀態。

用 git 歷史 revert

5 分鐘

HF dataset 也是 git repo。Push 前的 commit SHA 一定還在 history 裡，把 v3.0 tag 重新指回去就好：

python3 <<'EOF'
from huggingface_hub import HfApi, list_repo_commits
api = HfApi(token='YOUR_WRITE_TOKEN')
# 列最近的 commits
commits = list(list_repo_commits('Lebruhbruh/SARM-opendata-annotated-fixed', repo_type='dataset'))
for c in commits[:5]:
    print(f'{c.commit_id[:10]}  {c.created_at}  {c.title}')
# 確認哪個是 push 之前的，然後：
OLD_SHA = '貼上你想 revert 到的 commit SHA'
try: api.delete_tag('Lebruhbruh/SARM-opendata-annotated-fixed', tag='v3.0', repo_type='dataset')
except: pass
api.create_tag('Lebruhbruh/SARM-opendata-annotated-fixed', tag='v3.0',
               repo_type='dataset', revision=OLD_SHA)
EOF
不需要刪 commit — HF dataset 沒有「force push 砍歷史」這種事；只要把 v3.0 tag 移走， LeRobotDataset 就讀不到新版本，等同於 rollback。
